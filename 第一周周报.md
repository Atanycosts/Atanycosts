# 第一周周报
+ 题一
    > 数组末尾加一
    ```C++
    #include<iostream>
    using namespace std;
    #include<vector>
    //数组末位+1
    class Solution{
        public:
            void v(vector<int>&digits)
            {
                int n = digits.size();
                 for(int i =n-1;i <=0;i--)
                {
                    if(digits[i] == 9)
                {
                    digits[i] = 0;
                }
                else{
                    digits[i] +=1;
                    return;
                }
            }
        }
    };
    ```

+ 题三
  >将数组中的零元素,移动到末尾
  ```C++
  #include<iostream>
    #include<vector>
    using namespace std;
    //移动零元素   快慢指针
    class Solution{
        public:
            void z(vector<int>nums)
            {
                int n = nums.size(),left = 0,right =0;
                while(right<n)
                {
                    if(nums[right] !=0)
                    {
                        swap(nums[left],nums[right]);
                        left++;
                    }
                    else{
                        right++;
                    }
                }
            }
    /*
    01032
    10032
    13002
    12300
    */
    };
    ```

    >原地翻转字符串
    ```C++
    #include<iostream>
    using namespace std;
    #include<vector>
    //原地反转字符串
    class SD{
        public:
            void shuzu(vector<int>s ,int i)
            {
                int n =s.size();
                int temp;
                for(i = 0;i < n/2;i++){
                    temp =s[i];
                    s[i] =s[n-i-1];
                    s[n-i-1] =temp;
                }
            }
    ```
